public class SavingsAccount extends Account {
	final double minbal=5000.00;
	boolean withdraw(double cur_bal,double wdamt){
		if((cur_bal-wdamt)>minbal)
			return true;
		else
		return false;
	}
		

}

	final double ovrdrft_lmt=5000.00;

	}
}
